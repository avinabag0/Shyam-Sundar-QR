    // TRUE HORIZONTAL WHEEL CAROUSEL
    const carousel = document.getElementById("carousel");
    const cards = document.querySelectorAll(".carousel-card");
    const numCards = cards.length;
    const angleStep = 360 / numCards;
    let angle = 0;
    let startX = 0;
    let isDragging = false;
    let velocity = 0;
    let radius = 0;

    function updateRadius() {
      const width = window.innerWidth;
      const height = carousel.offsetHeight;
      radius = Math.min(width * 0.55, height * 0.75);
    }

    function updatePositions() {
      updateRadius();
      const centerX = carousel.offsetWidth / 2;
      const centerY = carousel.offsetHeight / 2;

      cards.forEach((card, i) => {
        const currentAngle = angle + (angleStep * i);
        const rad = (currentAngle * Math.PI) / 180;
        const x = Math.sin(rad) * radius;
        const z = Math.cos(rad) * radius - radius;

        const scale = 0.75 + (z + radius) / (2 * radius) * 0.5;
        const opacity = 0.5 + (z + radius) / (2 * radius) * 0.5;

        card.style.transform = `translateX(${x}px) translateZ(${z}px) scale(${scale}) rotateY(${-currentAngle}deg)`;
        card.style.opacity = opacity;
        card.style.zIndex = Math.round(z + radius);

        const isActive = Math.abs(z) < 100;
        card.classList.toggle("active", isActive);
      });
    }

    function smoothSnap() {
      const target = Math.round(angle / angleStep) * angleStep;
      const diff = target - angle;
      velocity = diff * 0.08;

      function animate() {
        velocity *= 0.92;
        angle += velocity;

        if (Math.abs(velocity) > 0.05) {
          updatePositions();
          requestAnimationFrame(animate);
        } else {
          angle = target;
          updatePositions();
        }
      }
      animate();
    }

    carousel.addEventListener("mousedown", (e) => {
      isDragging = true;
      startX = e.clientX;
      velocity = 0;
    });

    window.addEventListener("mouseup", () => {
      if (isDragging) {
        isDragging = false;
        smoothSnap();
      }
    });

    window.addEventListener("mousemove", (e) => {
      if (!isDragging) return;
      const delta = e.clientX - startX;
      startX = e.clientX;
      angle += delta * 0.25;
      velocity = delta * 0.25;
      updatePositions();
    });

    carousel.addEventListener("touchstart", (e) => {
      isDragging = true;
      startX = e.touches[0].clientX;
      velocity = 0;
    });

    window.addEventListener("touchend", () => {
      if (isDragging) {
        isDragging = false;
        smoothSnap();
      }
    });

    window.addEventListener("touchmove", (e) => {
      if (!isDragging) return;
      const delta = e.touches[0].clientX - startX;
      startX = e.touches[0].clientX;
      angle += delta * 0.25;
      velocity = delta * 0.25;
      updatePositions();
    });

    window.addEventListener("resize", updatePositions);
    updatePositions();
    setInterval(updatePositions, 16);

    // Modal
    const collections = {
      saree: [{title:'Silk Saree',img:'saree.jpg'},{title:'Designer',img:'saree.jpg'},{title:'Night View',img:'saree.jpg'}],
      bedding: [{title:'Bedsheets',img:'bedding.jpg'},{title:'Comforter',img:'bedding.jpg'},{title:'Setup',img:'bedding.jpg'}],
      textile: [{title:'Fabrics',img:'textile.jpg'},{title:'Wholesale',img:'textile.jpg'},{title:'Display',img:'textile.jpg'}],
      garments: [{title:'Fashion',img:'garments.jpg'},{title:'Ethnic',img:'garments.jpg'},{title:'Festive',img:'garments.jpg'}]
    };

    function openCategory(cat) {
      document.getElementById('modal-title').textContent = cat.charAt(0).toUpperCase() + cat.slice(1) + ' Store';
      const g = document.getElementById('gallery'); g.innerHTML = '';
      collections[cat].forEach(i => {
        g.innerHTML += `<div class="gallery-item"><img src="${i.img}" alt="${i.title}" loading="lazy"><div class="gallery-overlay"><div class="gallery-title">${i.title}</div></div></div>`;
      });
      document.getElementById('modal').classList.add('active');
      document.body.style.overflow = 'hidden';
      history.pushState({ modalOpen: true, category: cat }, '', `#${cat}`);
    }

    function closeModal() {
      document.getElementById('modal').classList.remove('active');
      document.body.style.overflow = 'auto';
      if (window.location.hash) history.back();
    }

    window.addEventListener('popstate', e => {
      if (e.state && e.state.modalOpen) openCategory(e.state.category);
      else closeModal();
    });

    document.addEventListener('keydown', e => { if (e.key === 'Escape') closeModal(); });
    document.getElementById('modal').addEventListener('click', e => { if (e.target.id === 'modal') closeModal(); });

    function scrollToCatalog() {
      document.getElementById('catalog').scrollIntoView({ behavior: 'smooth' });
    }
  </script>
